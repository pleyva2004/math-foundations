\subsection*{First, an example}

Before the formalism, a concrete picture. Let the true distribution be a fair
coin, $p = (0.5, 0.5)$ on $\{H, T\}$. Shannon's entropy is $H(p) = 1$ bit per
flip --- the optimal average code length.

Now suppose we mistakenly design our code for the biased belief
$q = (0.9, 0.1)$, assigning a short codeword to $H$ and a long one to $T$. The
average length we actually pay on data drawn from $p$ is the cross-entropy
\[
  H(p, q) = -0.5 \log_2 0.9 - 0.5 \log_2 0.1 \approx 0.076 + 1.66 \approx 1.74
  \text{ bits}.
\]
\emph{Read aloud:} ``H of p comma q equals minus one half log base two of zero
point nine minus one half log base two of zero point one, approximately one
point seven four bits.''
The cost of being wrong is the Kullback--Leibler divergence
$\KL(p \,\|\, q) = H(p, q) - H(p) \approx 0.74$ bits per symbol. This gap ---
extra bits paid for using the wrong model --- is exactly what cross-entropy
measures, and it is precisely the standard loss minimized in ML classification.

\section{Definitions}

\begin{definition}[Discrete cross-entropy]
Let $p, q$ be probability mass functions on a countable alphabet $\X$, with
$q(x) > 0$ whenever $p(x) > 0$. The \emph{cross-entropy} of $p$ with respect to
$q$ is
\[
  H(p, q) \;=\; -\sum_{x \in \X} p(x) \log q(x) \;=\; -\E_{X \sim p}[\log q(X)].
\]
\emph{Read aloud:} ``H of p comma q equals minus the sum over x in script X of
p of x times log q of x, which equals minus the expectation, X drawn from p, of
log q of X.''
We adopt $0 \log 0 = 0$. If there exists $x \in \X$ with $p(x) > 0$ and
$q(x) = 0$, then $H(p, q) = +\infty$.
\end{definition}

\begin{definition}[Continuous cross-entropy]
For probability densities $f_p, f_q$ on $\R^d$ with $f_q > 0$ on the support
of $f_p$,
\[
  H(p, q) \;=\; -\int_{\R^d} f_p(x) \log f_q(x) \, dx.
\]
\emph{Read aloud:} ``H of p comma q equals minus the integral over R-d of f-sub-p
of x times log f-sub-q of x, d x.''
\end{definition}

\begin{definition}[Kullback--Leibler divergence]
With the same hypotheses,
\[
  \KL(p \,\|\, q) \;=\; \sum_{x} p(x) \log \frac{p(x)}{q(x)} \;=\; H(p, q) - H(p),
\]
\emph{Read aloud:} ``D-K-L of p given q equals the sum over x of p of x times
log of p of x over q of x, which equals H of p comma q minus H of p.''
where $H(p) = -\sum_x p(x) \log p(x)$ is Shannon's entropy.
\end{definition}

\section{Gibbs' inequality}

\begin{lemma}[Logarithm bound]
\label{lem:logbound}
For all $z > 0$,
\[
  \ln z \;\le\; z - 1,
\]
\emph{Read aloud:} ``natural log of z is less than or equal to z minus one.''
with equality if and only if $z = 1$.
\end{lemma}

\begin{proof}
Let $f(z) = z - 1 - \ln z$ on $(0, \infty)$. Then $f'(z) = 1 - 1/z$, so $f'(z) < 0$
for $z \in (0, 1)$ and $f'(z) > 0$ for $z > 1$, with $f'(1) = 0$. Hence $f$ has a
strict global minimum at $z = 1$ with $f(1) = 0$. Therefore $f(z) \ge 0$
everywhere, with equality only at $z = 1$.
\end{proof}

\begin{theorem}[Gibbs' inequality]
\label{thm:gibbs}
For any two distributions $p, q$ on a common alphabet $\X$ with $q$ absolutely
continuous with respect to $p$ on the support of $p$,
\[
  H(p, q) \;\ge\; H(p),
\]
\emph{Read aloud:} ``H of p comma q is greater than or equal to H of p.''
with equality if and only if $p(x) = q(x)$ for all $x$ in the support of $p$.
\end{theorem}

\begin{proof}
We show the equivalent statement $\KL(p \,\|\, q) \ge 0$. Restrict the sum to
$\X_0 = \{x : p(x) > 0\}$; on this set $q(x) > 0$ by hypothesis, so
$q(x)/p(x)$ is a positive real number.

Apply Lemma~\ref{lem:logbound} with $z = q(x)/p(x)$ for each $x \in \X_0$:
\[
  \ln \frac{q(x)}{p(x)} \;\le\; \frac{q(x)}{p(x)} - 1.
\]
\emph{Read aloud:} ``natural log of q of x over p of x is less than or equal to
q of x over p of x, minus one.''
Multiplying both sides by $p(x) \ge 0$ preserves the inequality, and summing
over $x \in \X_0$ gives
\[
  \sum_{x \in \X_0} p(x) \ln \frac{q(x)}{p(x)}
  \;\le\;
  \sum_{x \in \X_0} \left( q(x) - p(x) \right)
  \;=\; \left( \sum_{x \in \X_0} q(x) \right) - 1
  \;\le\; 1 - 1 = 0,
\]
\emph{Read aloud:} ``the sum over x in script-X-zero of p of x times log of q of
x over p of x is at most the sum of q of x minus p of x, which equals the sum
of q of x minus one, which is at most one minus one, equals zero.''
where the last inequality uses $\sum_{x \in \X_0} q(x) \le \sum_{x \in \X} q(x) = 1$.
Converting to an arbitrary log base by dividing through by $\ln(\text{base})$ does
not change the sign, so
\[
  -\KL(p \,\|\, q) \;=\; \sum_{x \in \X_0} p(x) \log \frac{q(x)}{p(x)} \;\le\; 0,
\]
\emph{Read aloud:} ``minus D-K-L of p given q equals the sum over x in
script-X-zero of p of x log of q of x over p of x, which is less than or equal
to zero.''
i.e.\ $\KL(p \,\|\, q) \ge 0$. By the decomposition
$H(p, q) = H(p) + \KL(p \,\|\, q)$, this gives $H(p, q) \ge H(p)$.

\medskip
\emph{Equality case.} Suppose $H(p, q) = H(p)$, so $\KL(p \,\|\, q) = 0$.
Each summand $p(x) \ln(q(x)/p(x))$ is bounded above by $q(x) - p(x)$
(Lemma~\ref{lem:logbound}), and we must have equality in both the
Lemma-\ref{lem:logbound} step and the marginal step. Equality in
Lemma~\ref{lem:logbound} forces $q(x)/p(x) = 1$ for every $x \in \X_0$, i.e.\
$q(x) = p(x)$ on the support of $p$. The marginal step requires
$\sum_{x \in \X_0} q(x) = 1$, which is now automatic since
$\sum_{x \in \X_0} q(x) = \sum_{x \in \X_0} p(x) = 1$. Conversely, if $p = q$ on
$\X_0$ then every term in $\KL$ vanishes, so $H(p, q) = H(p)$.
\end{proof}

\begin{corollary}[Decomposition]
For any $p, q$ as above,
\[
  H(p, q) \;=\; H(p) + \KL(p \,\|\, q),
\]
\emph{Read aloud:} ``H of p comma q equals H of p plus D-K-L of p given q.''
and minimizing $H(p, q)$ over $q$ (with $p$ fixed) is equivalent to minimizing
$\KL(p \,\|\, q)$ over $q$.
\end{corollary}

\begin{proof}
Direct from the definitions: $-\sum_x p(x) \log q(x) = -\sum_x p(x) \log p(x) + \sum_x p(x) \log \frac{p(x)}{q(x)}$.
The second statement follows because $H(p)$ does not depend on $q$.
\end{proof}

\begin{example}[Fair coin, biased code]
Let $p = (\tfrac{1}{2}, \tfrac{1}{2})$ and $q = (0.9, 0.1)$ on $\X = \{0, 1\}$.
Then $H(p) = 1$ bit, and
\[
  H(p, q) = -\tfrac{1}{2} \log_2 0.9 - \tfrac{1}{2} \log_2 0.1 \approx 1.737 \text{ bits},
\]
\emph{Read aloud:} ``H of p comma q equals minus one half log base two of zero
point nine minus one half log base two of zero point one, approximately one
point seven three seven bits.''
so $\KL(p \,\|\, q) \approx 0.737$ bits --- the per-symbol coding overhead of
using a code matched to $q$ on data drawn from $p$.
\end{example}
